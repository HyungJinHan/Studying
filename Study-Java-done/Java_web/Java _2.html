Method의 기본 형식

 
public class Method_1 {
    
    public static void printA() {
        System.out.println("-");
        System.out.println("a");
        System.out.println("A");
    }
    // 다른 언어에서는 함수를 생성하는 것처럼 Method 생성
    // Alt+Shift+M 단축키로 변수화 가능
    // Method를 통해 값을 수정하면 printA()에 해당하는 모든 값이 수정
    
    public static void printB() {
        System.out.println("-");
        System.out.println("b");
        System.out.println("B");
    }
    
    private static void printC() {
        System.out.println("-");
        System.out.println("c");
        System.out.println("C");
    }
    
    public static void main(String[] args) {
    // 이 자체가 본문이라는 큰 Method
        
        System.out.println("Hello Method"); // Hello Method
        System.out.println(Math.floor(1.1)); // 1.0
        
        printA(); // Method 호출
        
        printB();
        
        printC();
 
    }
    
}
 
// 결과
// Hello Method
// 1.0
// -
// a
// A
// -
// b
// B
// -
// c
// C

Method의 Input (입력)

 
public class Method_1 {
    
    public static void main(String[] args) {
        printAll("hhj", "");
        printAll("hsh", "---");
        printAll("jhs", "===");
                // Argument (인자)
        // printAll()의 괄호 안의 값을 지정하면
        // Method의 text 부분이 지정한 값으로 수정됨
        // printAll(String name, String line)을 통해 여러 값을 지정 가능
    }
    
    public static void printAll(String name, String line) {
                                // Parameter (매개변수)
    // String(문자열)으로써 값은 text라는 이름으로 지정된 Method
        System.out.println(line);
        System.out.println(name);
        System.out.println(name);
    }
    
}
 
// 결과
// hhj
// hhj
// ---
// hsh
// hsh
// ===
// jhs
// jhs

Method의 Output (출력)

 
public class Method_output {
 
    public static void main(String[] args) {
                // void는 return값이 없다는 뜻
        System.out.println(str());
        System.out.println(num());
    }
    
    public static String str() {
    // return 값인 String(문자열)인 a를 a() 자리에 출력
        return "Welcome HHJ";
        // return은 Method가 종료됨
    }
    
    public static int num() {
    // return 값인 int(숫자)인 1를 a() 자리에 출력
        return 112233;
    }
 
}
 
// 결과
// Welcome HHJ
// 112233

-----Output의 활용-----

 
public class Method_1 {
    
    public static void main(String[] args) {
        System.out.println(printAll("hhj", ""));
        System.out.println(printAll("hsh", "---"));
        System.out.println(printAll("jhs", "==="));
    }
    
    public static String printAll(String name, String line) {
        String out = "";
        // out은 String이라는 것을 지정
        out = out + line + "\n";
        out = out + name + "\n";
        out = out + name;
        // printAll("name", "line")의 name과 line을 지정 시
        // 위의 정한 내용이 출력
        return out;
        // Method 종료
    }
    
}
 
// 결과
// hhj
// hhj
// ---
// hsh
// hsh
// ===
// jhs
// jhs

Access Level Modifiers


public 자리에 protected, default, private 등이 올 수 있음
↓                ↓
 
class Greeting {
    public static void Hi() {
        System.out.println("Hi!");
    }
}
 
class Greeting {
    protected static void Hi() {
        System.out.println("Hi!");
    }
}
 
class Greeting {
    default static void Hi() {
        System.out.println("Hi!");
    }
}
 
class Greeting {
    private static void Hi() {
        System.out.println("Hi!");
    }
}

Static


Static → Class Method
No Static → Instance Method
class Print {
    
    public String Line;
    
    public static void c(String Line) {
        System.out.println(Line);
        System.out.println("c");
        System.out.println("c");
    }
    
    public void b() {
    // Method가 Instance 소속일 경우 static을 지워줘야 함
        System.out.println(this.Line);
        System.out.println("b");
        System.out.println("b");
    }
 
    public void a() {
    // Method가 Instance 소속일 경우 static을 지워줘야 함
        System.out.println(this.Line);
        System.out.println("a");
        System.out.println("a");
    }
}
 
public class Static_ex {
    
    public static void main(String[] args) {
        
        // Instance
        // Print라는 class 안에 public String line으로 지정
        Print t1 = new Print();
        t1.Line = "---";
        t1.a();
        t1.b();
        // Print.a("---");
        // Print.b("---");
        
        Print t2 = new Print();
        t2.Line = "===";
        t2.a();
        t2.b();
        // Print.a("===");
        // Print.b("===");
        
        // class 소속의 c이므로 Method 생성 시 static 사용
        Print.c("~~~");
    }
}

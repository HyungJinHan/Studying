Inheritance(상속)의 기본 형식


 
class Cal {
    
    public int sum(int v1, int v2) {
        return v1+v2;
    }
    
    public int minus(int v1, int v2) {
        return v1-v2;
    }
}
 
class Cal2 extends Cal {
// extends Cal : Cal이라는 class를 상속받아 기능을 확장한다는 의미
// 별도의 method에 해당하는 코드 없이 작동 가능
    
}
 
public class inheritance_ex {
 
    public static void main(String[] args) {
        
        Cal c = new Cal();
        System.out.println(c.sum(5, 10)); // 15
        
        Cal c_ = new Cal();
        System.out.println(c_.minus(10, 5)); // 5
        
        Cal2 c2 = new Cal2();
        System.out.println(c2.sum(50, 100)); // 150
        
        Cal2 c2_ = new Cal2();
        System.out.println(c2_.minus(100, 50)); // 50
 
    }
 
}

기능의 개선 & 발전

 
class Cal {
    
    public int sum(int v1, int v2) {
        return v1+v2;
    }
    
    public int minus(int v1, int v2) {
        return v1-v2;
    }
}
 
class Cal2 extends Cal {
// extends Cal : Cal이라는 class를 상속받아 기능을 확장한다는 의미
// 별도의 method에 해당하는 코드 없이 작동 가능
    
    public int multi(int v1, int v2) {
        return v1*v2;
    }
    // 상속받은 method 외의 추가적인 method를 입력해서 확장 가능
    
    public int minus(int v1, int v2) {
        System.out.println("Changed!");
        return v2-v1;
    }
    // Override(ing) (덮어쓰기)
    // 상속받은 값이어도 덮어쓰기와 같이 수정 가능
    
}
 
public class inheritance_ex {
 
    public static void main(String[] args) {
        
        Cal c = new Cal();
        System.out.println(c.sum(5, 10)); // 15
        System.out.println(c.minus(10, 5)); // 5
        
        Cal2 c2 = new Cal2();
        System.out.println(c2.sum(50, 100)); // 150
        System.out.println(c2.minus(100, 50)); // Changed! / -50
        System.out.println(c2.multi(50, 100)); // 5000
        
 
    }
 
}

Overriding VS Overloading


Overriding → 상속받은 method 외의 다른 method의 추가 개념
Overloading → 출력값의 문제가 없는 과적재의 개념 (인자 2개에서 3개로 추가)
 
class Cal {
    
    public int sum(int v1, int v2) {
        return v1+v2;
    }
    
    public int sum(int v1, int v2, int v3) {
        return v1+v2+v3;
    }
    // Overload(ing) (과적재)
    // 같은 코드라고 하더라도 인자, 출력값을 추가하는 작업
    
    public int minus(int v1, int v2) {
        return v1-v2;
    }
}
 
class Cal2 extends Cal {
// extends Cal : Cal이라는 class를 상속받아 기능을 확장한다는 의미
// 별도의 method에 해당하는 코드 없이 작동 가능
    
    public int multi(int v1, int v2) {
        return v1*v2;
    }
    // 상속받은 method 외의 추가적인 method를 입력해서 확장 가능
    
    public int minus(int v1, int v2) {
        System.out.println("Changed!");
        return v2-v1;
    }
    // Override(ing) (덮어쓰기)
    // 상속받은 값이어도 덮어쓰기와 같이 수정 가능
    
}
 
public class inheritance_ex {
 
    public static void main(String[] args) {
        
        Cal c = new Cal();
        System.out.println(c.sum(5, 10)); // 15
        System.out.println(c.sum(5, 10, 100)); // 115
        // Overload(ing)(과적재)의 결과
        // 인자의 수가 3개인 method를 찾아 값을 출력
        System.out.println(c.minus(10, 5)); // 5
        
        Cal2 c2 = new Cal2();
        System.out.println(c2.sum(50, 100)); // 150
        System.out.println(c2.sum(50, 100, 150)); // 300
        // 상속에 따른 Overload(ing)(과적재)의 결과
        // 인자의 수가 3개인 상속받은 method를 찾아 값을 출력
        System.out.println(c2.minus(100, 50)); // Changed!, -50
        // 상속값의 Override(ing)(덮어쓰기)의 결과
        System.out.println(c2.multi(50, 100)); // 5000
        // Override(ing)(덮어쓰기)를 통한 method 추가의 결과
        
 
    }
 
}

Super & This


Super(부모의 값) → 해당 method가 부모의 method의 값으로 지정됨
This(자신의 값) → 해당 method가 자기 자신의 내부의 method의 값으로 지정됨
 
class Cal {
    
    public int sum(int v1, int v2) {
        return v1+v2+v2+v1+v1+v1+v2;
    }
    
    public int sum(int v1, int v2, int v3) {
        return v1+v2+v3;
    }
    
    public int minus(int v1, int v2) {
        return v1-v2;
    }
}
 
class Cal2 extends Cal {
    
    public int multi(int v1, int v2) {
        return v1*v2;
    }
    // this를 통해 multi1의 multi(v1, v2)에 대한 method로써 정의됨
    
    public int multi1(int v1, int v2) {
        return this.multi(v1, v2)*v2;
    }
    // this 사용의 예시 (multi는 상속받은 값이 아닌 override한 값)
    
    public int minus(int v1, int v2) {
        System.out.println("Changed!");
        return v2-v1;
    }
    
    public int sum(int v1, int v2) {
        return super.sum(v1, v2);
    }
    
    public int sum(int v1, int v2, int v3) {
        return v1+v2+v3;
    }
    
}
 
public class inheritance_ex {
 
    public static void main(String[] args) {
        
        Cal c = new Cal();
        System.out.println(c.sum(5, 10)); // 15
        System.out.println(c.sum(5, 10, 100)); // 115
        System.out.println(c.minus(10, 5)); // 5
        
        Cal2 c2 = new Cal2();
        System.out.println(c2.sum(50, 100)); // 150 → super 사용으로 500
        // super 사용을 통해 부모의 v1+v2+v2+v1+v1+v1+v2라는 별도의 method 입력없이 쉽게 상속받아 사용 가능
        System.out.println(c2.sum(50, 100, 150)); // 300
        System.out.println(c2.minus(100, 50)); // Changed!, -50
        System.out.println(c2.multi(50, 100)); // 5000
        System.out.println(c2.multi1(50, 100)); // this 사용으로 500000
        // (v1, v2)*v2에 대해 this 사용을 통해 Cal2 내부적으로 가지고 있는 v1*v2의 값을 (v1, v2)에 대입시켜
        // 나온 값에 추가적으로 v2를 곱함
        
 
    }
 
}

상속 & 생성자

 
class Cal {
 
    int v1, v2;
    
    Cal (int v1, int v2) {
        this.v1 = v1; this.v2 = v2;
    }
    //생성자 생성
    
    public int minus() {
        return v1-v2;
    }
 
}
 
class Cal2 extends Cal {
 
    // int v1, v2;
    // 위의 값은 아래의 super를 통해 부모의 값을 가져왔기 때문에 제거
    
    Cal2(int v1, int v2) {
        super(v1, v2);
        // Cal (int v1, int v2) {this.v1 = v1; this.v2 = v2;}를 의미
        // 부모 class의 생성자를 super로 가져와야 부모 class의 생성자 사용이 가능
    }
    
    public int multi() {
        return this.v1*v2;
    }
    
}
 
public class inheritance_ex {
 
    public static void main(String[] args) {
        
        Cal c = new Cal(10, 100);
        System.out.println(c.minus()); // -90
        // return this.v1-v2의 값의 결과
        
        Cal2 c2 = new Cal2(10, 100);
        System.out.println(c2.multi()); // 1000
        // return this.v1*v2의 값의 결과
 
    }
 
}

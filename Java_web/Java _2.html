Method의 기본 형식

 
public class Method_1 {
    
    public static void printA() {
        System.out.println("-");
        System.out.println("a");
        System.out.println("A");
    }
    // 다른 언어에서는 함수를 생성하는 것처럼 Method 생성
    // Alt+Shift+M 단축키로 변수화 가능
    // Method를 통해 값을 수정하면 printA()에 해당하는 모든 값이 수정
    
    public static void printB() {
        System.out.println("-");
        System.out.println("b");
        System.out.println("B");
    }
    
    private static void printC() {
        System.out.println("-");
        System.out.println("c");
        System.out.println("C");
    }
    
    public static void main(String[] args) {
    // 이 자체가 본문이라는 큰 Method
        
        System.out.println("Hello Method"); // Hello Method
        System.out.println(Math.floor(1.1)); // 1.0
        
        printA(); // Method 호출
        
        printB();
        
        printC();
 
    }
    
}
 
// 결과
// Hello Method
// 1.0
// -
// a
// A
// -
// b
// B
// -
// c
// C
